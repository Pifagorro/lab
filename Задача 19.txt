#include <stdio.h>

#define max(a,b) ((a) > (b) ? (a) : (b))
#define min(a,b) ((a) < (b) ? (a) : (b))

int main()
{
	int i, j, k;
	int count = 0;

	for (i = 0; i <= 13; i++) {
		k = 0;
		for (j = max(0, i - 18); j <= min(9, i); j++) {
			k += min(9, i - j) - max(0, i - j - 9) + 1;
		}
		count += k * k;
	}

	count *= 2;
	printf("Щасливих квитків: %d\n", count);
	return 0;
}
