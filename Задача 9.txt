#include <iostream>
#include <stdio.h>
int temp, i, n,k,j;
int m[101];
using namespace std;
int main(void)
{
	system("chcp 1251");
	cout << "Введіть кількість елементів массива";
	scanf("%d", &n);
	cout << "Введіть k";
	scanf("%d", &k);
	for (i = 0; i < n; i++)
		scanf("%d", &m[i]);
	for (j = 0; j < k; j++) {
		temp = m[n - 1];
		for (i = n - 1; i > 0; i--)
			m[i] = m[i - 1];
		m[0] = temp;
	}
	for (i = 0; i < n; i++)
		printf("%d ", m[i]);
	printf("\n");
	return 0;
}
