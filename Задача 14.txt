#include <iostream>
#include <vector>
#include <stack>
using namespace std;
int main() {
	system("chcp 1251");
	size_t n;
	cout << "¬вед≥ть N:" <<endl;
	cin >> n;
	vector <int> a(n);
	cout << "¬вед≥ть N чисел: "  << endl;
	for (size_t i = 0; i < a.size(); i++)
		cin >> a[i];
	stack <int> st;
	for (size_t i = a.size(); i > 0; i--) {
		while (st.size() > 0 && st.top() <= a[i - 1])
			st.pop();
		int t = a[i - 1];
		a[i - 1] = st.size() > 0 ? st.top() : 0;
		st.push(t);
	}
	for (size_t i = 0; i < a.size(); i++)
		cout << a[i] << ' ';
	cout << '\n';
}
